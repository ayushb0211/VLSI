VHDL Code for 8-bit ALU

library IEEE;
use IEEE.STD_LOGIC_1164.ALL;
use IEEE.STD_LOGIC_ARITH.ALL;
use IEEE.STD_LOGIC_UNSIGNED.ALL;

entity alu_lab8  is
    Port (A,B : in  STD_LOGIC_VECTOR (7 downto 0);
           sel : in  STD_LOGIC_VECTOR (2 downto 0);
           Y : out  STD_LOGIC_VECTOR (7 downto 0));
end alu_lab8;

architecture Behavioral of alu_lab8  is                                                                                                        
begin
process(A,B,sel)
  begin
     case sel is
 when "000" => Y <=A+B;
 when "001" => Y <=A-B;
 when "010" => Y <=A AND B;
 when "011" => Y <=A NAND B;
 when "100" => Y <=A XOR B;
 when "101" => Y <=A XNOR B;
 when "110" => Y <=A OR B;
 when "111" => Y <=A;
 when others => Y <="ZZZZZZZZ";
    end case;
end process;
end Behavioral;
